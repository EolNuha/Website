26       def run(self):
27           run = True
28   
29           while run:
30               for event in pygame.event.get():
31                   if event.type == pygame.QUIT:
32                       run = False
33   
34                   if event.type == pygame.MOUSEBUTTONUP:
35                       # check if hit start btn

38                       if self.btn[0] <= x <= self.btn[0] + self.btn[2]:
39                           if self.btn[1] <= y <= self.btn[1] + self.btn[3]:

44                           if self.btn1[1] <= y <= self.btn1[1] + self.btn1[3]:
45                               game2 = Game2(self.win)

50                               game3 = Game3(self.win)
51                               game3.run()
